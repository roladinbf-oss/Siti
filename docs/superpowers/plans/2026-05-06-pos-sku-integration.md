# POS SKU Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Link POS system SKUs to internal recipes so sales from the cash register can be attributed to the correct recipe, enabling accurate food-cost tracking and unmatched-sale alerting.

**Architecture:** Add `sku` to `recipes` table and a new `pos_products` catalog table (populated by Excel import). The frontend gets: an import button on the recipes page, SKU badge in the recipe list, a SKU autocomplete field in the recipe edit modal, sales attribution at save-time, and two dashboard counters (unlinked recipes, unmatched sales). All DB access goes through the existing `sbGet/sbPost/sbPatch` helpers hitting `/api/:table` on Railway.

**Tech Stack:** PostgreSQL (Railway), Express `api.js` CRUD layer, vanilla JS ES6+, SheetJS (already loaded via `loadSheetJS()`), single `index.html`.

```js
async function handlePosSkuExcel(input) {
  if(!input.files||!input.files.length) return;
  const file = input.files[0]; input.value='';
  showToast('מעבד קובץ מק"טים...','warn');
  try {
    if(!window.XLSX) await loadSheetJS();
    const buf = await file.arrayBuffer();
    const wb = XLSX.read(buf,{type:'array'});
    const ws = wb.Sheets[wb.SheetNames[0]];
    const rows = XLSX.utils.sheet_to_json(ws,{header:1,defval:''});
    if(!rows.length){showToast('הקובץ ריק','err');return;}

    // Flexible header detection — Hebrew & English variants
    const header = rows[0].map(h=>String(h).trim());
    const skuI = header.findIndex(h=>
      /מק"ט|מקט|sku|barcode|קוד פריט|קוד מוצר|item.*code|product.*code/i.test(h)
    );
    const nameI = header.findIndex(h=>
      /שם פריט|שם מוצר|product.*name|item.*name|תיאור|description/i.test(h)
    );
    const catI = header.findIndex(h=>
      /קטגוריה|category|סוג/i.test(h)
    );
    if(skuI<0){showToast('לא נמצאה עמודת מק"ט בקובץ','err');return;}
    if(nameI<0){showToast('לא נמצאה עמודת שם מוצר בקובץ','err');return;}

    const parsed = [];
    rows.slice(1).forEach(cells=>{
      const sku = String(cells[skuI]||'').trim();
      const product_name = String(cells[nameI]||'').trim();
      if(!sku||!product_name) return;
      const category = catI>=0 ? String(cells[catI]||'').trim() : null;
      parsed.push({sku, product_name, category});
    });

    if(!parsed.length){showToast('לא נמצאו שורות תקינות','err');return;}

    // Upsert: fetch existing SKUs then insert-or-update
    const existing = await sbGet('pos_products','?limit=5000') || [];
    const existingMap = {};
    existing.forEach(p=>{ existingMap[p.sku]=p; });

    let added=0, updated=0;
    for(const row of parsed) {
      if(existingMap[row.sku]) {
        // Update if name/category changed
        const ex = existingMap[row.sku];
        if(ex.product_name!==row.product_name || ex.category!==row.category) {
          await sbPatch('pos_products','sku=eq.'+encodeURIComponent(row.sku),{
            product_name: row.product_name,
            category: row.category,
            imported_at: new Date().toISOString()
          });
          updated++;
        }
      } else {
        await sbPost('pos_products',{
          sku: row.sku,
          product_name: row.product_name,
          category: row.category||null,
          imported_at: new Date().toISOString()
        });
        added++;
      }
    }

    POS_PRODUCTS = await sbGet('pos_products','?order=product_name.asc&limit=2000') || [];
    renderRecipes();
    showToast('יובאו '+parsed.length+' מוצרים | חדשים: '+added+' | עודכנו: '+updated+' ✓');
  } catch(e) {
    showToast('שגיאה: '+e.message.slice(0,50),'err');
  }
}
```

```js
function openSkuDrop() {
  const q = (document.getElementById('recSku').value||'').trim().toLowerCase();
  const drop = document.getElementById('recSkuDrop');
  if(!q || !POS_PRODUCTS.length) { drop.style.display='none'; return; }
  const matches = POS_PRODUCTS.filter(p=>
    String(p.sku).toLowerCase().includes(q) ||
    (p.product_name||'').toLowerCase().includes(q)
  ).slice(0,12);
  if(!matches.length){ drop.style.display='none'; return; }
  drop.innerHTML = matches.map(p=>
    '<div style="padding:8px 12px;cursor:pointer;border-bottom:1px solid var(--border);font-size:13px;" '+
    'onmousedown="selectPosSku(\''+p.sku+'\',\''+p.product_name.replace(/'/g,"\\'")+'\')" '+
    'onmouseover="this.style.background=\'var(--s2)\'" onmouseout="this.style.background=\'\'">'+
      '<span style="font-weight:700;color:var(--gold);">'+p.sku+'</span> — '+p.product_name+
      (p.category?'<span style="font-size:11px;color:var(--text3);"> ('+p.category+')</span>':'')+
    '</div>'
  ).join('');
  drop.style.display='block';
}

function selectPosSku(sku, name) {
  // Check if this SKU is already linked to another recipe
  const conflict = RECIPES.find(r=>r.sku===sku && r.id!==_currentRecId);
  if(conflict) {
    if(!confirm('מק"ט '+sku+' כבר משויך למתכון "'+conflict.name+'".\nלשייך בכל זאת?')) return;
  }
  document.getElementById('recSku').value = sku;
  document.getElementById('recSkuDrop').style.display='none';
  document.getElementById('recSkuSuggestion').style.display='none';
}

function applySkuSuggestion(sku) {
  selectPosSku(sku, '');
}

function findPosFuzzyMatch(recipeName) {
  if(!POS_PRODUCTS.length) return null;
  const clean = s => s.replace(/[^֐-׿a-zA-Z0-9 ]/g,'').toLowerCase();
  const rWords = clean(recipeName).split(' ').filter(w=>w.length>=3);
  let best = null, bestScore = 0;
  POS_PRODUCTS.forEach(p=>{
    const pWords = clean(p.product_name||'').split(' ').filter(w=>w.length>=3);
    const shared = rWords.filter(w=>pWords.some(pw=>pw.includes(w)||w.includes(pw))).length;
    const score = shared / Math.max(rWords.length, pWords.length, 1);
    if(score > bestScore && score >= 0.6) { bestScore=score; best=p; }
  });
  return best;
}
```
